	trigger = {

		name = ccw.71.a

		}

		custom_effect_tooltip = THIS_IS_HISTORICAL_TT
		activate_mission = CHI_disarmament_mission

			limit = {
				is_ai = yes
			    check_variable = { Root.num_battalions > 2499 }

		}

			limit = {
				is_ai = yes
			    check_variable = { Root.num_battalions > 2999 }

            # disband 25% of the army of the current tag
            set_variable = { disband_fraction = 0.25 }
            disband_units_fraction = yes
            clear_variable = disband_fraction
		}
		if = {
			limit = {
				is_ai = yes
			    check_variable = { Root.num_battalions > 3499 }
			}
            # disband 30% of the army of the current tag
            set_variable = { disband_fraction = 0.3 }
            disband_units_fraction = yes
            clear_variable = disband_fraction
		}
		if = {
