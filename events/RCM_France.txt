			is_puppet = no
		}

			is_fully_controlled_by = ROOT
			is_core_of = FRA
		}
		NOT = {
			tag = FRA
			is_ai = no
		}
	}
	mean_time_to_happen = {
		days = 2
	}
	option = {
		name = rcmfrance.8.a
		ai_chance = { factor = 100 }
		effect_tooltip = {
			every_owned_state = {
				limit = {
					is_fully_controlled_by = ROOT
					is_core_of = FRA
				}
				FRA = {
					transfer_state = PREV
				}
			}
		}
		FRA = {
			country_event = rcmfrance.9
		}
	} 
}
