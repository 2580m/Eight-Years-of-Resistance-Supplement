											monument_upgrade_trigger = yes
										}
									}
								}
								add_to_temp_array = { alert_monuments = monument_id }
								add_to_temp_array = { alert_monuments_states = THIS.id }
							}
						}
					}
				}
			}
			else = {
				all_owned_state = {
					if = {
						limit = {
							check_variable = { alert_monuments^num < 6 }
							state_has_any_monument = yes
							is_controlled_by = ROOT
							any_of = {
								array = state_monuments_array
								set_temp_variable = { monument_id = v }
								OR = {
									AND = {
										monument_build_potential_trigger = yes
										monument_build_trigger = yes
									}
									AND = {
										monument_upgrade_potential_trigger = yes
										monument_upgrade_trigger = yes
									}
								}
							}
						}
						all_of = {
							array = state_monuments_array
							set_temp_variable = { monument_id = v }
							if = {
								limit = {
									check_variable = { alert_monuments^num < 6 }
									OR = {
										AND = {
											monument_build_potential_trigger = yes
											monument_build_trigger = yes
										}
										AND = {
											monument_upgrade_potential_trigger = yes
											monument_upgrade_trigger = yes
										}
									}
								}
								add_to_temp_array = { alert_monuments = monument_id }
								add_to_temp_array = { alert_monuments_states = THIS.id }
							}
						}
					}
				}
			}
			set_temp_variable = { index_temp = -1 }
		}
		localization_key = "[This.GetMonumentConstructionAlertMonumentsListEntry]"
	}
	text = {
		trigger = {
			always = yes
		}
		localization_key = ""
	}
}
